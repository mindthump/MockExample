10   """ 
11   The whole point of mocking is to monitor and/or alter the _behavior_ 
12   of running objects **without changing the original code**. Patch as 
13   "small" as you can; whenever possible only the specific behavior that 
14   is actually _used_. If a resource class is expensive or impossible to 
15   instantiate you may need to patch the whole class. 
16   """

19   @patch('student.PeopleDatabase.get_name_by_id') 
20   def test_student(mock_student_getname): 
21       """ 
22       A single method is the simplest to patch. The student.py module 
23       imports PersonDataSource directly so that's what we call it. The 
24       tricky part is knowing its name inside the running code; 'import' 
25       statements are a good clue. 
26       """ 
27    
28       # Set the value our mock object will return for the mocked 
29       # get_name_by_id() function The id argument is immaterial - since 
30       # we override the return value, the input doesn't go anywhere. 
31       # "Sam" never appears in the data, so it also acts as a guard from 
32       # accidental false-positive tests. 
33       mock_student_getname.return_value = "Sam" 
34    
35       # We can see instance_pds but not _pds or module_pds 
36       # Yet, it does't matter: we can still mock its method by name 
37       name = Student(2).name() 
38       assert name == "Sam"

41   @patch('people_data.PeopleDatabase.get_name_by_id') 
42   def test_employee(mock_employee_getname): 
43       """ 
44       The same test, except the target name of the mocked method 
45       """ 
46       mock_employee_getname.return_value = "Bob" 
47       employee_name = employee.Employee(1).name() 
48    
49       # NOTE: Why don't we assert == "Bob"? 
50       # We switched the name to "Bob" when Employee asked the database 
51       # for it, but we didn't change what Employee did with it after 
52       # we switched it. In this case, it applied formatting akin to a 
53       # __repl__() then returned that, so that's what we test against. 
54       assert employee_name == "#1 - Bob"

64   def test_context_manager(): 
65       """ 
66       'patch' can be used in a context manager ('with...'). This style is 
67       good when you only want to patch a function during part of a test 
68       """ 
69    
70       # No patch here, works normally 
71       name = employee.Employee(1).name() 
72       assert name == "#1 - Alice" 
73    
74       with patch('employee.people_data.PeopleDatabase.get_name_by_id') as mock_employee_datasource_getname: 
75           # Set a side-effect for our mock object. If it is an iterable 
76           # each call will return the next value. It could call a function, 
77           # taking the original arguments. 
78    
79           mock_employee_datasource_getname.side_effect = ['Bob', 'Tom'] 
80    
81           _employee = employee.Employee(16) 
82           employee_name = _employee.name() 
83           assert employee_name == "#16 - Bob" 
84    
85           # Notice that it doesn't matter that we re-instantiate the 
86           # Student object, because we've patched the method - not the class. 
87           _employee = employee.Employee(105) 
88           # This is the second time we are running the patched method 
89           employee_name = _employee.name() 
90           assert employee_name == "#105 - Tom"

93   @patch('employee.people_data.PeopleDatabase') 
94   def test_class_patch(mock_datasource_class): 
95       """ 
96       This example is for patching an entire CLASS. Most of the time you 
97       only need to patch a specific method! 
98       """ 
99    
100      # The return value from directly calling a class constructor is an 
101      # instance of the class. Because the patch above is for a whole 
102      # class, the return_value is a new mock object acting in place of 
103      # the instantiated object, just like any other call to the mock 
104      # object. On the mocked 'instance', the get_name_by_id method is 
105      # also a mock... 
106      mock_ds_instance = mock_datasource_class.return_value 
107      mock_instance_getname_function = mock_ds_instance.get_name_by_id 
108      mock_instance_getname_function.return_value = "Bob" 
109   
110      # mock_datasource_class.return_value.get_name_by_id.return_value = "Bob" 
111   
112      test_employee = employee.Employee(1) 
113      employee_name = test_employee.name() 
114      assert employee_name == "#1 - Bob"
